# Enable USB OTG (to support USB host mode)
dtoverlay=dwc2
dr_mode=host
dtparam=usb=on


# Set HDMI output to 720p
hdmi_group=1
hdmi_mode=4

# Enable I2C
dtparam=i2c_arm=on

# Enable SPI
dtparam=spi=on

# Enable 1-Wire
dtoverlay=w1-gpio

# Enable UART for RS485
dtoverlay=uart3
dtoverlay=uart5

# Enable SPI and configure CAN bus
dtparam=spi=on
dtoverlay=mcp2515-can0,oscillator=16000000,interrupt=25
